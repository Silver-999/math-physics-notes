\section*{Advanced Mathematics(Volume One)}
\subsection*{Theory}
\begin{gather*}
\lim_{n \to \infty} x_n = a \leftrightarrow \forall \varepsilon > 0, \, \exists N \in \mathbb{N}^*, \, n > N, \, |x_n - a| < \varepsilon
\end{gather*}
\begin{enumerate}[label=\arabic*., leftmargin=*, align=left, noitemsep]
\item Removable discontinuity: \[\lim\limits_{x \to 1} \frac{x^2 - 1}{x - 1}\]
\item Jump discontinuity: \[\lim\limits_{x \to 0} \begin{cases} x - 1, & x < 0 \\ 0, & x = 0 \\ x + 1, & x > 0 \end{cases}\]
\item Infinite discontinuity: \[\lim\limits_{x \to \frac{\pi}{2}} \tan x\]
\item Oscillating discontinuity: \[\lim\limits_{x \to 0} \sin \frac{1}{x}\]
\end{enumerate}
\begin{gather*}
\lim_{\alpha} \frac{\beta}{\alpha} = 0 \leftrightarrow \beta = o(\alpha),\quad \sec x \cos x = 1,\quad \csc x \sin x = 1,\quad \tan^2 x + 1 = \sec^2 x \\
[f^{-1}(x)]' = \lim_{\Delta x \to 0} \frac{\Delta y}{\Delta x} = \lim_{\Delta x \to 0} \frac{1}{\frac{\Delta x}{\Delta y}};\quad
\text{eg: } y' = (\arcsin x)' = \frac{1}{(\sin y)'} = \frac{1}{\cos y} = \frac{1}{\sqrt{1 - \sin^2 y}} = \frac{1}{\sqrt{1 - x^2}} \\
(\arcsin x)' = \frac{1}{\sqrt{1 - x^2}},\quad (\arccos x)' = -\frac{1}{\sqrt{1 - x^2}},\quad (\arctan x)' = \frac{1}{1 + x^2},\quad (\operatorname{arccot} x)' = -\frac{1}{1 + x^2}
\end{gather*}

\textbf{Rolle's Theorem:} 
$f(x)$ is continuous on $[a,b]$, derivable on $(a,b)$, and $f(a) = f(b)$.
\begin{equation*}
\exists\, \xi \in (a,b), \quad f'(\xi) = 0
\end{equation*}

\textbf{Lagrange's Theorem:} 
$f(x)$ is continuous on $[a,b]$ and derivable on $(a,b)$.
\begin{equation*}
\exists\, \xi \in (a,b), \quad \frac{f(b) - f(a)}{b - a} = f'(\xi)
\end{equation*}

\textbf{Cauchy's Theorem:} 
$f(x)$ and $F(x)$ are continuous on $[a,b]$ and derivable on $(a,b)$.
\begin{equation*}
\exists\, \xi \in (a,b), \quad \frac{f(b) - f(a)}{F(b) - F(a)} = \frac{f'(\xi)}{F'(\xi)}
\end{equation*}

\textbf{Taylor's Theorem:}
\begin{equation*}
f(x) = f(x_0) + f'(x_0)(x - x_0) + \ldots + \frac{f^{(n)}(x_0)}{n!} (x - x_0)^n + o((x - x_0)^n)
\end{equation*}

\textbf{Maclaurin's Theorem:}
\begin{equation*}
f(x) = f(0) + f'(0)x + \ldots + \frac{f^{(n)}(0)}{n!} x^n + o(x^n)
\end{equation*}

\textbf{Concave function:} 
$f''(x) > 0$:
\begin{equation*}
f\left(\frac{x_1 + x_2}{2}\right) < \frac{f(x_1) + f(x_2)}{2}
\end{equation*}

\textbf{Convex function:} 
$f''(x) < 0$:
\begin{equation*}
f\left(\frac{x_1 + x_2}{2}\right) > \frac{f(x_1) + f(x_2)}{2}
\end{equation*}

\subsection*{Integrals}
\begin{gather*}
\int \tan x \, dx = -\ln |\cos x| + C, \quad \int \cot x \, dx = \ln |\sin x| + C \\
\int \sec x \, dx = \ln |\sec x + \tan x| + C, \quad \int \csc x \, dx = \ln |\csc x - \cot x| + C \\
\int \sec^2 x \, dx = \tan x + C, \quad \int \csc^2 x \, dx = -\cot x + C \\
\int \sec x \tan x \, dx = \sec x + C, \quad \int \csc x \cot x \, dx = -\csc x + C \\
\int \frac{dx}{x^2 + a^2} = \frac{1}{a} \arctan \frac{x}{a} + C \quad (a > 0), \quad \int \frac{dx}{\sqrt{x^2 \pm a^2}} = \ln \left|x + \sqrt{x^2 \pm a^2}\right| + C \quad (a > 0) \\
\int \frac{dx}{\sqrt{a^2 - x^2}} = \arcsin \frac{x}{a} + C \quad (a > 0), \quad \int \frac{dx}{a^2 - x^2} = \frac{1}{2a} \ln \left|\frac{a + x}{a - x}\right| + C \quad (a > 0) \\[6pt]
\end{gather*}

\begin{flushleft}
Order: arc, log, power, exp, trig: $\int u(x) dv(x) = u(x)v(x) - \int v(x) du(x)$ \\
Trigonometric substitution: $u = \tan \frac{x}{2}$, $\sin x = \frac{2u}{1 + u^2}$, $\cos x = \frac{1 - u^2}{1 + u^2}$, $dx = \frac{2}{1 + u^2} du$ \\
Newton-Leibniz formula: $\int_a^b f'(x) dx = f(b) - f(a)$, $\left[\int_{\phi_1(x)}^{\phi_2(x)} f(t) dt\right]' = \phi_2'(x) \cdot f[\phi_2(x)] - \phi_1'(x) \cdot f[\phi_1(x)]$ \\
Polar area1: $\rho = \rho(\theta)$, $\theta_1 = \alpha$, $\theta_2 = \beta$, $S = \frac{1}{2} \int_\alpha^\beta \rho^2(\theta) d\theta$ \\
Polar area2: $\rho_1 = \rho_1(\theta)$, $\rho_2 = \rho_2(\theta)$, $\theta_1 = \alpha$, $\theta_2 = \beta$, $S = \frac{1}{2} \int_\alpha^\beta [\rho_2^2(\theta) - \rho_1^2(\theta)] d\theta$ \\
Volume of revolution: $V_x = \pi \int_a^b f^2(x) dx$, $V_y = 2\pi \int_a^b x f(x) dx$, $L = \int_\alpha^\beta \sqrt{1 + f'^2(x)} dx$ \\
Wallis formula: $\int_0^{\pi/2} \sin^n x dx = \int_0^{\pi/2} \cos^n x dx = 
\begin{cases}
\frac{(n-1)!!}{n!!} \cdot \frac{\pi}{2}, & n \geq 2 \text{ even} \\ 
\frac{(n-1)!!}{n!!}, & n \geq 3 \text{ odd}
\end{cases}$
\end{flushleft}

\subsection*{Differential Equations}
Separable variable: $\frac{dy}{dx} = f(x)g(y) \Rightarrow \int \frac{dy}{g(y)} = \int f(x) dx$ \\
Homogeneous equation: $\frac{dy}{dx} = g\left(\frac{y}{x}\right) \quad \text{Let } u = \frac{y}{x} \Rightarrow \frac{dy}{dx} = u + x \frac{du}{dx} \Rightarrow \int \frac{du}{g(u) - u} = \int \frac{dx}{x}$ \\
First order linear differential equation: $\frac{dy}{dx} + P(x)y = Q(x) \Rightarrow y = e^{-\int P(x) dx} \left[\int Q(x) e^{\int P(x) dx} dx + C\right]$ \\
Bernoulli equation: $y' + p(x)y = q(x)y^\alpha \quad \text{Let } z = y^{1-\alpha} \Rightarrow z' + (1-\alpha)p(x)z = (1-\alpha)q(x) \quad (\alpha \neq 0,1)$ \\
Second order constant coefficient: $y'' + py' + qy = 0$ \\
Characteristic equation: $\mu^2 + p\mu + q = 0$ \\
\begin{itemize}[leftmargin=*, noitemsep]
\item Distinct real roots $\mu_1 \neq \mu_2$: $y = C_1 e^{\mu_1 x} + C_2 e^{\mu_2 x}$
\item Repeated real root $\mu_1 = \mu_2$: $y = (C_1 + C_2 x) e^{\mu_1 x}$
\item Complex roots $\mu_{1,2} = \alpha \pm \beta i$: $y = e^{\alpha x}(C_1 \cos \beta x + C_2 \sin \beta x)$
\end{itemize}

\subsection*{Directional Derivative}
\begin{gather}
D_{\vv{u}}f(x_0,y_0,z_0) = \lim_{t \to 0} \frac{f(x_0 + t\alpha, y_0 + t\beta, z_0 + t\gamma) - f(x_0,y_0,z_0)}{t}
\end{gather}

\subsection*{Definition of Gradient}
\begin{gather}
\nabla f = \left( \frac{\partial f}{\partial x}, \frac{\partial f}{\partial y}, \frac{\partial f}{\partial z} \right) = \frac{\partial f}{\partial x}\vv{i} + \frac{\partial f}{\partial y}\vv{j} + \frac{\partial f}{\partial z}\vv{k}
\end{gather}

\subsection*{Polar Coordinates}
\begin{gather}
\iint\limits_{D} f(x,y) \, dA = \iint\limits_{D'} f(r\cos\theta, r\sin\theta) \, r \, dr d\theta
\end{gather}

\subsection*{Spherical Coordinates}
\begin{gather}
\iiint\limits_{\Omega} f(x,y,z) \, dV = \iiint\limits_{\Omega'} f(r\sin\phi\cos\theta, r\sin\phi\sin\theta, r\cos\phi) \, r^2 \sin\phi \, dr d\phi d\theta
\end{gather}

\subsection*{First Type of Line Integral}
\begin{gather}
\int_L f(x,y,z) \, ds = \int_\alpha^\beta f(x(t),y(t),z(t)) \sqrt{\left(\frac{dx}{dt}\right)^2 + \left(\frac{dy}{dt}\right)^2 + \left(\frac{dz}{dt}\right)^2} \, dt
\end{gather}

\subsection*{Second Type of Line Integral}
\begin{gather}
\int_L \vv{F} \cdot d\vv{r} = \int_L Pdx + Qdy + Rdz = \int_\alpha^\beta \left[P\frac{dx}{dt} + Q\frac{dy}{dt} + R\frac{dz}{dt}\right] dt
\end{gather}

\subsection*{First Type of Surface Integral}
\begin{gather}
\iint_\Sigma f(x,y,z) \, dS = \iint_{D_{xy}} f(x,y,z(x,y)) \sqrt{1 + \left(\frac{\partial z}{\partial x}\right)^2 + \left(\frac{\partial z}{\partial y}\right)^2} \, dxdy
\end{gather}

\subsection*{Second Type of Surface Integral}
\begin{gather}
\iint_\Sigma \vv{F} \cdot d\vv{S} = \iint_\Sigma P dydz + Q dzdx + R dxdy = \iint_{D_{xy}} \left[P\left(-\frac{\partial z}{\partial x}\right) + Q\left(-\frac{\partial z}{\partial y}\right) + R\right] dxdy
\end{gather}

\subsection*{Ratio Test / Root Test}
\begin{gather}
L = \lim_{n \to \infty} \left| \frac{a_{n+1}}{a_n} \right| = \lim_{n \to \infty} \sqrt[n]{|a_n|} 
\end{gather}

\subsection*{Leibniz Test for Alternating Series}
\begin{gather}
\text{If } \sum_{n=1}^{\infty} (-1)^{n-1} a_n \text{ with } a_n > 0 \text{ satisfies: } a_{n+1} \leq a_n \text{ and } \lim_{n \to \infty} a_n = 0, \text{ then the series converges.}
\end{gather}

\subsection*{Power Series Expansion of Common Functions}
\setcounter{equation}{0}
\begin{align}
B_{2n} &= (-1)^{n-1} \frac{2(2n)!}{(2\pi)^{2n}} \zeta(2n) \\
E_{2n} &= (-1)^n \frac{2^{2n+2}(2n)!}{\pi^{2n+1}} \beta(2n+1) \\
\sin x &= \sum_{n=0}^\infty \frac{(-1)^n x^{2n+1}}{(2n+1)!} = x - \frac{x^3}{6} + \frac{x^5}{120} - \cdots, \quad x \in (-\infty, +\infty) \\
\cos x &= \sum_{n=0}^\infty \frac{(-1)^n x^{2n}}{(2n)!} = 1 - \frac{x^2}{2} + \frac{x^4}{24} - \cdots, \quad x \in (-\infty, +\infty) \\
\tan x &= \sum_{n=1}^\infty \frac{B_{2n} (-4)^n (1-4^n)}{(2n)!} x^{2n-1} = x + \frac{x^3}{3} + \frac{2x^5}{15} + \cdots, \quad |x| < \frac{\pi}{2} \\
\cot x &= \sum_{n=0}^\infty \frac{(-1)^n 2^{2n} B_{2n}}{(2n)!} x^{2n-1} = \frac{1}{x} - \frac{x}{3} - \frac{x^3}{45} - \cdots, \quad 0 < |x| < \pi \\
\sec x &= \sum_{n=0}^\infty \frac{(-1)^n E_{2n}}{(2n)!} x^{2n} = 1 + \frac{x^2}{2} + \frac{5x^4}{24} + \cdots, \quad |x| < \frac{\pi}{2} \\
\csc x &= \sum_{n=0}^\infty \frac{(-1)^{n+1} 2(2^{2n-1}-1) B_{2n}}{(2n)!} x^{2n-1} = \frac{1}{x} + \frac{x}{6} + \frac{7x^3}{360} + \cdots, \quad 0 < |x| < \pi \\
\arcsin x &= \sum_{n=0}^\infty \frac{(2n)!}{4^n (n!)^2 (2n+1)} x^{2n+1} = x + \frac{x^3}{6} + \frac{3x^5}{40} + \cdots, \quad |x| \leq 1 \\
\arccos x &= \frac{\pi}{2} - \arcsin x = \frac{\pi}{2} - \sum_{n=0}^\infty \frac{(2n)!}{4^n (n!)^2 (2n+1)} x^{2n+1}, \quad |x| \leq 1 \\
\arctan x &= \sum_{n=0}^\infty \frac{(-1)^n}{2n+1} x^{2n+1} = x - \frac{x^3}{3} + \frac{x^5}{5} - \cdots, \quad |x| \leq 1 \\
\cot^{-1} x &= \frac{\pi}{2} - \arctan x = \frac{\pi}{2} - \sum_{n=0}^\infty \frac{(-1)^n}{2n+1} x^{2n+1}, \quad |x| \leq 1 \\
\sec^{-1} x &= \arccos\left(\frac{1}{x}\right) = \frac{\pi}{2} - \sum_{n=0}^\infty \frac{(2n)!}{4^n (n!)^2 (2n+1)} x^{-2n-1}, \quad |x| \geq 1 \\
\csc^{-1} x &= \arcsin\left(\frac{1}{x}\right) = \sum_{n=0}^\infty \frac{(2n)!}{4^n (n!)^2 (2n+1)} x^{-2n-1}, \quad |x| \geq 1 \\
e^x &= \sum_{n=0}^\infty \frac{x^n}{n!} = 1 + x + \frac{x^2}{2} + \frac{x^3}{6} + \cdots, \quad x \in (-\infty, +\infty) \\
\frac{1}{1-x} &= \sum_{n=0}^\infty x^n = 1 + x + x^2 + x^3 + \cdots, \quad x \in (-1, 1) \\
\frac{1}{1+x} &= \sum_{n=0}^\infty (-1)^n x^n = 1 - x + x^2 - x^3 + \cdots, \quad x \in (-1, 1) \\
\ln(1+x) &= \sum_{n=1}^\infty \frac{(-1)^{n-1} x^n}{n} = x - \frac{x^2}{2} + \frac{x^3}{3} - \cdots, \quad x \in (-1, 1] \\
-\ln(1-x) &= \sum_{n=1}^\infty \frac{x^n}{n} = x + \frac{x^2}{2} + \frac{x^3}{3} + \cdots, \quad x \in [-1, 1) \\
\frac{1}{2}\ln\left(\frac{1+x}{1-x}\right) &= \sum_{n=0}^\infty \frac{x^{2n+1}}{2n+1} = x + \frac{x^3}{3} + \frac{x^5}{5} + \cdots, \quad x \in (-1, 1) \\
(1+x)^\alpha &= 1 + \sum_{n=1}^\infty \frac{\alpha(\alpha-1)\cdots(\alpha-n+1)}{n!} x^n, \quad x \in (-1, 1)
\end{align}
